javascript:(function(){    /* CONFIGURACIÓN: Velocidad (1500 = 1,5 segundo). Súbelo si se traban. */    var VELOCIDAD = 1500;    /* 1. VERIFICAR SI ESTAMOS EN EL SRI */    if (window.location.href.indexOf("srienlinea.sri.gob.ec") === -1) {        if(confirm("🌐 No estás en el SRI. ¿Ir a Facturación Electrónica?")){            window.open("https://srienlinea.sri.gob.ec/tu-portal-internet/accederAplicacion.jspa?redireccion=SI&idGrupo=55&idServicio=328", "_blank");        }        return;    }    /* 2. LIMPIEZA Y CREACIÓN DEL BOTÓN */    var idBtn = "btn_sri_final";    var antiguo = document.getElementById(idBtn);    if(antiguo) antiguo.remove();    var btn = document.createElement("button");    btn.id = idBtn;    btn.innerHTML = "📥 DETECTAR XMLs";    /* Estilo visual para que no se pierda */    btn.style.cssText = "position:fixed; top:85px; right:20px; z-index:999999; padding:12px 20px; background-color:#2e7d32; color:white; font-family:Arial, sans-serif; font-weight:bold; font-size:14px; border:2px solid white; border-radius:8px; box-shadow:0 4px 10px rgba(0,0,0,0.5); cursor:pointer;";    document.body.appendChild(btn);    /* 3. LÓGICA DE DESCARGA */    btn.onclick = function(){        /* Buscar todos los elementos que parezcan XML (imágenes o enlaces) */        var encontrados = [];        var imagenes = document.querySelectorAll("img[src*='xml'], img[src*='XML']");                imagenes.forEach(function(img){            /* Si la imagen está dentro de un link, usamos el link. Si no, la imagen. */            if(img.parentElement.tagName === 'A') encontrados.push(img.parentElement);            else encontrados.push(img);        });        /* Si no hay imágenes, buscar texto */        if(encontrados.length === 0){            document.querySelectorAll("a").forEach(function(a){                if(a.title && a.title.toUpperCase().includes("XML")) encontrados.push(a);            });        }        /* --- VALIDACIONES --- */        if(encontrados.length === 0){            alert("❌ No veo facturas.\n\n1. Dale a 'Consultar'.\n2. Espera que cargue la tabla.\n3. Vuelve a tocar este botón.");            return;        }        var total = encontrados.length;        if(!confirm("✅ ENCONTRADOS " + total + " XMLs.\n\nSe guardarán automáticamente en tu carpeta configurada.\n\n¿EMPEZAR?")) return;        /* --- INICIO DEL BUCLE --- */        var i = 0;        btn.onclick = null; /* Bloquear botón para no darle doble clic */        btn.style.backgroundColor = "#ff8f00"; /* Color Naranja */        var intervalo = setInterval(function(){            if(i >= total){                clearInterval(intervalo);                btn.innerHTML = "✅ FIN";                btn.style.backgroundColor = "blue";                alert("¡Proceso Terminado!");                /* Opcional: Remover botón al finalizar */                setTimeout(function(){ btn.remove(); }, 4000);                return;            }            /* Acción de clic */            try {                btn.innerHTML = "⏳ " + (i+1) + " / " + total;                encontrados[i].click();                /* Borde rojo para marcar visualmente */                if(encontrados[i].style) encontrados[i].style.border = "2px solid red";            } catch(e) {                console.log("Error en fila " + i);            }            i++;        }, VELOCIDAD);    };})();
